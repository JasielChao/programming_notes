    <script>
        /* Scripts para el Tab Circular Movil */
        document.addEventListener("DOMContentLoaded", () => {
            const items_tab = document.querySelectorAll('.my_ul_tab_circular li');
            const my_btn_plus = document.querySelector('.my_btn_plus');
            const my_tabs_content = document.querySelectorAll('.my_tabs_content');

            my_btn_plus.addEventListener("click", () => {
                const items_tab = document.querySelectorAll('.my_ul_tab_circular li');
                var contador = 50;
                if (my_btn_plus.classList.contains("active")) {
                    my_btn_plus.classList.remove('active');
                    for (i = 0; i < items_tab.length; i++) {
                        items_tab[i].style.transform = "translate( 0px, 0px)";
                    }
                } else {
                    my_btn_plus.classList.add('active');
                    for (i = 0; i < items_tab.length; i++) {
                        if (i % 2 == 1) {
                            items_tab[i].style.transform = "translate(" + contador + "px, 20px)";
                        } else {
                            items_tab[i].style.transform = "translate(" + contador + "px, -20px)";
                        }
                        contador += 50;
                    }
                }

            });

            for (i = 0; i < items_tab.length; i++) {
                items_tab[i].addEventListener("click", cambiar_tab(i));

            }


            function cambiar_tab(index) {
                return function() {
                    for (j = 0; j < my_tabs_content.length; j++) {
                        if (my_tabs_content[j].classList.contains("active")) {
                            my_tabs_content[j].classList.remove("active");
                        }
                    }
                    my_tabs_content[index].classList.add("active");
                }
            }


        });
    </script>
